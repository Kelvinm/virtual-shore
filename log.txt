#### `app/events.py`

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from .schemas import EventCreate, EventUpdate, EventResponse
from .dependencies import get_db
from .crud import create_event, update_event, delete_event

router = APIRouter()

@router.post("/", response_model=EventResponse)
def create_new_event(event: EventCreate, db: Session = Depends(get_db)):
    return create_event(db=db, event=event)

@router.put("/{event_id}", response_model=EventResponse)
def update_event_by_id(event_id: int, event: EventUpdate, db: Session = Depends(get_db)):
    updated_event = update_event(db=db, event_id=event_id, event=event)
    if not updated_event:
        raise HTTPException(status_code=404, detail="Event not found")
    return updated_event

@router.delete("/{event_id}", response_model=EventResponse)
def delete_event_by_id(event_id: int, db: Session = Depends(get_db)):
    deleted_event = delete_event(db=db, event_id=event_id)
    if not deleted_event:
        raise HTTPException(status_code=404, detail="Event not found")
    return deleted_event
```

#### `app/holidays.py`

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from .schemas import HolidayResponse
from .dependencies import get_db
from .crud import get_holiday

router = APIRouter()

@router.get("/{country}/{date}", response_model=HolidayResponse)
def check_holiday(country: str, date: str, db: Session = Depends(get_db)):
    holiday = get_holiday(db=db, country=country, date=date)
    if not holiday:
        return {"is_holiday": False}
    return {"is_holiday": True}
```

#### `app/schemas.py`

```python
from pydantic import BaseModel

class EventBase(BaseModel):
    start_date: str
    end_date: str
    title: str
    description: str

class EventCreate(EventBase):
    pass

class EventUpdate(EventBase):
    id: int

class EventResponse(EventBase):
    id: int
    user_id: int

class HolidayResponse(BaseModel):
    is_holiday: bool
```

#### `app/crud.py`

```python
from sqlalchemy.orm import Session
from .models import Event, Holiday

def create_event(db: Session, event: EventCreate):
    db_event = Event(**event.dict())
    db.add(db_event)
    db.commit()
    db.refresh(db_event)
    return db_event

def update_event(db: Session, event_id: int, event: EventUpdate):
    db_event = db.query(Event).filter(Event.id == event_id).first()
    if not db_event:
        return None
    for key, value in event.dict(exclude_unset=True).items():
        setattr(db_event, key, value)
    db.commit()
    db.refresh(db_event)
    return db_event

def delete_event(db: Session, event_id: int):
    db_event = db.query(Event).filter(Event.id == event_id).first()
    if not db_event:
        return None
    db.delete(db_event)
    db.commit()
    return db_event

def get_holiday(db: Session, country: str, date: str):
    holiday = db.query(Holiday).filter_by(country=country, date=date).first()
    return holiday is not None
```

#### `app/models.py`

```python
from sqlalchemy import Column, Integer, String, Date
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()

class Event(Base):
    __tablename__ = "events"
    id = Column(Integer, primary_key=True)
    user_id = Column(Integer, nullable=False)
    start_date = Column(Date, nullable=False)
    end_date = Column(Date, nullable=False)
    title = Column(String, nullable=False)
    description = Column(String)

class Holiday(Base):
    __tablename__ = "holidays"
    id = Column(Integer, primary_key=True)
    country = Column(String, nullable=False)
    date = Column(Date, nullable=False)
    name = Column(String)
```
